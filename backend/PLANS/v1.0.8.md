```python
class DatabaseConfig:
    """数据库配置管理器"""
    
    @staticmethod
    def get_database_url() -> str:
        """根据环境变量获取数据库 URL"""
        db_type = os.getenv("DB_TYPE", "sqlite").lower()
        
        if db_type == "postgresql":
            host = os.getenv("DB_HOST", "localhost")
            port = os.getenv("DB_PORT", "5432")
            user = os.getenv("DB_USER", "btdeck")
            password = os.getenv("DB_PASSWORD", "")
            database = os.getenv("DB_NAME", "btdeck")
            
            return (
                f"postgresql+asyncpg://{user}:{password}"
                f"@{host}:{port}/{database}"
            )
        else:
            # SQLite（默认）
            db_path = os.getenv("DB_PATH", "config/app.db")
            return f"sqlite:///{db_path}"
    
    @staticmethod
    def get_engine_config():
        """获取引擎配置"""
        db_type = os.getenv("DB_TYPE", "sqlite").lower()
        
        if db_type == "postgresql":
            return {
                "pool_size": 10,
                "max_overflow": 20,
                "pool_pre_ping": True,
                "echo": False
            }
        else:
            return {
                "echo": False,
                "connect_args": {"check_same_thread": False}
            }

# 创建引擎
engine = create_async_engine(
    DatabaseConfig.get_database_url(),
    **DatabaseConfig.get_engine_config()
)
```

### Phase 2: 连接池管理 (2天)

#### ✅ 任务2.1: 创建连接池管理器
**文件**: `app/services/downloader_pool.py`

```python
class BtpManagerPool:
    """下载器连接池"""
    
    def __init__(self, max_connections: int = 10):
        self._connections = {}  # {instance_id: (client, last_used, health)}
        self._semaphore = asyncio.Semaphore(max_connections)
        self._lock = asyncio.Lock()
        self._health_check_interval = 60
        
    async def get_client(self, instance_id: int) -> Optional[BtpManager]:
        """获取或创建客户端"""
        
        # 信号量限制并发
        async with self._semaphore:
            if instance_id in self._connections:
                client, last_used, health = self._connections[instance_id]
                
                # 健康检查
                if health:
                    return client
                else:
                    # 移除失效连接
                    await self.remove_client(instance_id)
            
            # 创建新连接
            client = await self._create_client(instance_id)
            
            if client:
                self._connections[instance_id] = (client, time.time(), True)
            
            return client
    
    async def health_check_all(self) -> Dict[int, bool]:
        """健康检查所有连接"""
        results = {}
        
        for instance_id, (client, last_used, _) in list(self._connections.items()):
            try:
                is_healthy = await self._health_check(client)
                self._connections[instance_id] = (client, last_used, is_healthy)
                results[instance_id] = is_healthy
            except Exception:
                results[instance_id] = False
        
        return results
    
    async def _health_check(self, client: BtpManager) -> bool:
        """检查客户端健康状态"""
        try:
            if client.downloader_type == 0:  # qBittorrent
                version = await asyncio.to_thread(client.client.app_version)
                return bool(version)
            elif client.downloader_type == 1:  # Transmission
                stats = await asyncio.to_thread(client.client.session_stats)
                return bool(stats)
        except Exception:
            return False
```

**验收标准**:
- [ ] 支持并发限制（Semaphore）
- [ ] 支持连接健康检查
- [ ] 支持自动移除失效连接

### Phase 3: 迁移与部署 (1-2天)

#### ✅ 任务3.1: 数据迁移脚本
**文件**: `scripts/migrate_to_postgresql.sh`

```bash
#!/bin/bash

echo "🔄 Migrating from SQLite to PostgreSQL..."

# 1. 备份 SQLite 数据
echo "📦 Backing up SQLite database..."
python -c "
from app.database import SessionLocal
from app.models import *
import json

db = SessionLocal()
downloaders = db.query(BtDownloaders).all()
torrents = db.query(TorrentInfo).all()

with open('backup.json', 'w') as f:
    json.dump({
        'downloaders': [d.__dict__ for d in downloaders],
        'torrents': [t.__dict__ for t in torrents]
    }, f, default=str)

print(f'Backed up {len(downloaders)} downloaders, {len(torrents)} torrents')
"

# 2. 创建 PostgreSQL 数据库
echo "🔧 Creating PostgreSQL database..."
PGPASSWORD=$DB_PASSWORD psql -h $DB_HOST -U $DB_USER -c "CREATE DATABASE $DB_NAME;"

# 3. 运行迁移
echo "🚀 Running migrations..."
alembic upgrade head

# 4. 导入数据
echo "📥 Importing data to PostgreSQL..."
python scripts/import_data.py

echo "✅ Migration complete!"
```
