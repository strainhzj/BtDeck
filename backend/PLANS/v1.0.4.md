```python
class TorrentStateManager:
    """种子状态管理器 - 动静数据分离"""
    
    def __init__(self):
        # 静态数据缓存（10分钟更新）
        self.static_cache = {}  # {torrent_hash: TorrentStaticInfo}
        self.static_cache_time = None
        
        # 动态数据缓存（10秒更新）
        self.dynamic_cache = {}  # {torrent_hash: TorrentDynamicInfo}
        self.dynamic_cache_time = None
        
    async def get_torrents_with_speed(
        self,
        downloader_id: str,
        skip: int = 0,
        limit: int = 50
    ) -> Dict[str, Any]:
        """获取种子列表（包含实时速度）"""
        
        # 1. 获取静态数据
        static_data = await self._get_static_data(downloader_id, skip, limit)
        
        # 2. 获取动态数据
        dynamic_data = await self._get_dynamic_data(downloader_id)
        
        # 3. 合并数据
        merged = []
        for torrent in static_data["list"]:
            dynamic = dynamic_data.get(torrent["hash"], {})
            merged.append({
                **torrent,
                "downloadSpeed": dynamic.get("downloadSpeed", 0),
                "uploadSpeed": dynamic.get("uploadSpeed", 0),
                "num_seeds": dynamic.get("num_seeds", 0),
                "num_leechs": dynamic.get("num_leechs", 0),
                "speed_update_time": dynamic.get("update_time")
            })
        
        return {
            "code": "success",
            "data": {
                "total": static_data["total"],
                "list": merged
            }
        }
```

#### ✅ 任务1.2: 速度 API 接口
**文件**: `app/api/endpoints/torrent_speed.py`

```python
from fastapi import APIRouter

router = APIRouter(prefix="/torrents", tags=["torrents"])

@router.get("/speed-all")
async def get_all_torrents_speed():
    """获取所有种子的实时速度（轻量级接口）"""
    
    cached_downloaders = await app.state.store.get_snapshot()
    
    result = []
    for downloader in cached_downloaders:
        if downloader.fail_time > 0:
            continue
        
        try:
            # 获取传输信息（仅速度数据）
            transfer_info = await asyncio.to_thread(downloader.client.transfer_info)
            
            # 获取种子级别的速度数据
            torrents_data = await asyncio.to_thread(
                downloader.client.torrents_info,
                limit=500
            )
            
            torrent_speeds = []
            for torrent in torrents_data:
                torrent_speeds.append({
                    "hash": torrent.hash,
                    "downloadSpeed": torrent.dlspeed / 1024,
                    "uploadSpeed": torrent.upspeed / 1024,
                    "num_seeds": torrent.num_seeds,
                    "num_leechs": torrent.num_leechs
                })
            
            result.append({
                "downloader_id": downloader.downloader_id,
                "downloader_name": downloader.nickname,
                "dl_info_speed": transfer_info.get("dl_info_speed", 0) / 1024,
                "up_info_speed": transfer_info.get("up_info_speed", 0) / 1024,
                "torrents": torrent_speeds,
                "last_update": time.time()
            })
        except Exception as e:
            logger.warning(f"获取下载器速度失败: {e}")
    
    return {
        "code": "success",
        "msg": "获取速度数据成功",
        "data": result
    }
```

#### ✅ 任务2.2: 种子列表组件改造
**文件**: `BtDeck_fronted/src/views/torrents/index.vue`（改造现有组件，非新建）

> **注意**: 项目使用 Vue 2.6.12 Options API，禁止使用 Composition API。
> 现有组件已包含完整的种子列表功能（高级搜索、批量操作），仅需在此基础上添加速度轮询。

```vue
<script lang="ts">
import { Component, Vue } from 'vue-property-decorator'
import { getTorrentList, getActiveTorrents } from '@/api/torrents'

@Component({})
export default class TorrentList extends Vue {
  // 现有数据（保持不变）
  tableData: any[] = []
  listQuery = { page: 1, pageSize: 20, total: 0 }
  loading = false

  // 新增：速度轮询相关
  speedTimer: number | null = null
  activeHashSpeedMap: Record<string, any> = {}
  pendingUpdate = false

  // 合并后的种子列表（computed）
  get torrentsWithSpeed() {
    return this.tableData.map((torrent: any) => {
      const dynamic = this.activeHashSpeedMap[torrent.hash]
      if (!dynamic) return torrent
      return {
        ...torrent,
        downloadSpeed: dynamic.downloadSpeed ?? torrent.downloadSpeed ?? 0,
        uploadSpeed: dynamic.uploadSpeed ?? torrent.uploadSpeed ?? 0
      }
    })
  }

  // 1秒轮询：仅获取活跃种子速度
  async loadActiveSpeed() {
    if (this.pendingUpdate) return
    const vm = this
    vm.pendingUpdate = true
    try {
      const res = await getActiveTorrents()
      if (res.code === '200') {
        const map: Record<string, any> = {}
        ;(res.data || []).forEach((t: any) => {
          map[t.hash] = t
        })
        vm.activeHashSpeedMap = map
      }
    } catch (e) {
      console.error('速度轮询失败:', e)
    } finally {
      vm.pendingUpdate = false
    }
  }

  startSpeedPolling() {
    this.loadActiveSpeed()
    this.speedTimer = window.setInterval(() => {
      this.loadActiveSpeed()
    }, 1000)
  }

  stopSpeedPolling() {
    if (this.speedTimer) {
      clearInterval(this.speedTimer)
      this.speedTimer = null
    }
  }

  mounted() {
    this.loadData()
    this.startSpeedPolling()
  }

  beforeDestroy() {
    this.stopSpeedPolling()
  }
}
</script>
```
