### Phase 1: 路径发现增强 (1天)

#### ✅ 任务1.1: 新增 btdeck 路径发现
**文件**: `app/tasks/scheduler/downloader_path_scan.py`

**修改内容**:
```python
class DownloaderPathScanTask:
    """下载器路径扫描任务（增强版）"""
    
    async def _discover_btdeck_paths(self) -> List[str]:
        """发现 btdeck 配置的路径
        
        Returns:
            List[str]: btdeck 外部路径列表
        """
        btdeck_paths = set()
        
        # 1. 从 path_mapping 配置获取 external 路径
        try:
            from app.database import AsyncSessionLocal
            from app.models import PathMapping
            
            async with AsyncSessionLocal() as db:
                result = await db.execute(
                    select(PathMapping.external_path)
                    .distinct()
                )
                btdeck_paths.update([row[0] for row in result if row[0]])
        except Exception as e:
            logger.warning(f"获取 path_mapping 失败: {e}")
        
        # 2. 从 path_mapping_rules 转换后的路径
        try:
            from app.services.path_mapping_service import PathMappingService
            mapping_service = PathMappingService()
            
            async with AsyncSessionLocal() as db:
                result = await db.execute(
                    select(PathMappingRule)
                )
                rules = result.scalars().all()
                
                for rule in rules:
                    internal_paths = await self._get_internal_paths_for_rule(rule)
                    
                    for internal_path in internal_paths:
                        external_path = mapping_service.internal_to_external(internal_path)
                        if os.path.isabs(external_path):
                            btdeck_paths.add(external_path)
        except Exception as e:
            logger.warning(f"获取 path_mapping_rules 失败: {e}")
        
        # 3. 从历史路径转移记录获取
        try:
            from app.models import PathTransferHistory
            
            async with AsyncSessionLocal() as db:
                result = await db.execute(
                    select(PathTransferHistory.target_path)
                    .distinct()
                    .order_by(PathTransferHistory.transfer_time.desc())
                    .limit(100)
                )
                btdeck_paths.update([row[0] for row in result if row[0]])
        except Exception as e:
            logger.warning(f"获取历史路径转移记录失败: {e}")
        
        return list(btdeck_paths)
```

#### ✅ 任务1.2: 修改扫描任务主逻辑
**文件**: `app/tasks/scheduler/downloader_path_scan.py`

**修改内容**:
```python
async def execute(self, **kwargs):
    """执行路径扫描任务（增强版）"""
    from app.tasks.logger import TaskLogger
    
    async with TaskLogger("downloader_path_scan", task_type=1) as logger:
        try:
            # 1. 获取下载器路径（原有逻辑）
            downloader_paths = await self._discover_downloader_paths()
            logger.add_log(f"下载器路径: {len(downloader_paths)} 个")
            
            # 2. ✅ 新增: 获取 btdeck 路径
            btdeck_paths = await self._discover_btdeck_paths()
            logger.add_log(f"btdeck路径: {len(btdeck_paths)} 个")
            
            # 3. 合并所有路径
            all_paths = downloader_paths + btdeck_paths
            logger.add_log(f"总扫描路径: {len(all_paths)} 个")
            
            # 4. 去重
            unique_paths = list(set(all_paths))
            logger.add_log(f"去重后路径: {len(unique_paths)} 个")
            
            # 5. 扫描路径
            scan_results = await self._scan_paths(unique_paths, logger)
            
            # 6. 保存结果
            await self._save_scan_results(scan_results, logger)
            
            return {
                "status": "success",
                "total_paths": len(unique_paths),
                "scan_results": len(scan_results)
            }
            
        except Exception as e:
            logger.add_log(f"路径扫描失败: {e}")
            raise
```

**验收标准**:
