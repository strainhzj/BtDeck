#### ✅ 任务1.1: 创建状态管理器
**文件**: `app/services/torrent_state_manager.py`

```python
class TorrentStateManager:
    """种子状态管理器 - 动静数据分离"""
    
    def __init__(self):
        # 静态数据缓存（10分钟更新）
        self.static_cache = {}  # {torrent_hash: TorrentStaticInfo}
        self.static_cache_time = None
        
        # 动态数据缓存（10秒更新）
        self.dynamic_cache = {}  # {torrent_hash: TorrentDynamicInfo}
        self.dynamic_cache_time = None
        
    async def get_torrents_with_speed(
        self,
        downloader_id: str,
        skip: int = 0,
        limit: int = 50
    ) -> Dict[str, Any]:
        """获取种子列表（包含实时速度）"""
        
        # 1. 获取静态数据
        static_data = await self._get_static_data(downloader_id, skip, limit)
        
        # 2. 获取动态数据
        dynamic_data = await self._get_dynamic_data(downloader_id)
        
        # 3. 合并数据
        merged = []
        for torrent in static_data["list"]:
            dynamic = dynamic_data.get(torrent["hash"], {})
            merged.append({
                **torrent,
                "downloadSpeed": dynamic.get("downloadSpeed", 0),
                "uploadSpeed": dynamic.get("uploadSpeed", 0),
                "num_seeds": dynamic.get("num_seeds", 0),
                "num_leechs": dynamic.get("num_leechs", 0),
                "speed_update_time": dynamic.get("update_time")
            })
        
        return {
            "code": "success",
            "data": {
                "total": static_data["total"],
                "list": merged
            }
        }
```

#### ✅ 任务1.2: 速度 API 接口
**文件**: `app/api/endpoints/torrent_speed.py`

```python
from fastapi import APIRouter

router = APIRouter(prefix="/torrents", tags=["torrents"])

@router.get("/speed-all")
async def get_all_torrents_speed():
    """获取所有种子的实时速度（轻量级接口）"""
    
    cached_downloaders = await app.state.store.get_snapshot()
    
    result = []
    for downloader in cached_downloaders:
        if downloader.fail_time > 0:
            continue
        
        try:
            # 获取传输信息（仅速度数据）
            transfer_info = await asyncio.to_thread(downloader.client.transfer_info)
            
            # 获取种子级别的速度数据
            torrents_data = await asyncio.to_thread(
                downloader.client.torrents_info,
                limit=500
            )
            
            torrent_speeds = []
            for torrent in torrents_data:
                torrent_speeds.append({
                    "hash": torrent.hash,
                    "downloadSpeed": torrent.dlspeed / 1024,
                    "uploadSpeed": torrent.upspeed / 1024,
                    "num_seeds": torrent.num_seeds,
                    "num_leechs": torrent.num_leechs
                })
            
            result.append({
                "downloader_id": downloader.downloader_id,
                "downloader_name": downloader.nickname,
                "dl_info_speed": transfer_info.get("dl_info_speed", 0) / 1024,
                "up_info_speed": transfer_info.get("up_info_speed", 0) / 1024,
                "torrents": torrent_speeds,
                "last_update": time.time()
            })
        except Exception as e:
            logger.warning(f"获取下载器速度失败: {e}")
    
    return {
        "code": "success",
        "msg": "获取速度数据成功",
        "data": result
    }
```

#### ✅ 任务2.2: 种子列表组件优化
**文件**: `BtDeck_fronted/src/views/dashboard/TorrentList.vue`

```vue
<template>
  <el-table :data="torrentsWithSpeed" v-loading="loading" stripe>
    <!-- 静态数据列 -->
    <el-table-column prop="name" label="名称" min-width="200" />
    <el-table-column prop="size" label="大小" width="120" />
    <el-table-column prop="progress" label="进度" width="120">
      <template #default="{ row }">
        <el-progress :percentage="row.progress || 0" />
      </template>
    </el-table-column>
    
    <!-- 动态数据列（实时速度） -->
    <el-table-column prop="downloadSpeed" label="下载速度" width="120">
      <template #default="{ row }">
        <span :class="getSpeedClass(row.downloadSpeed)">
          {{ formatSpeed(row.downloadSpeed || 0) }}
        </span>
      </template>
    </el-table-column>
    
    <el-table-column prop="uploadSpeed" label="上传速度" width="120">
      <template #default="{ row }">
        <span :class="getSpeedClass(row.uploadSpeed)">
          {{ formatSpeed(row.uploadSpeed || 0) }}
        </span>
      </template>
    </el-table-column>
  </el-table>
</template>

<script lang="ts">
import { defineComponent, ref, computed, onMounted, onUnmounted } from 'vue'
import { getTorrents, getTorrentsSpeedAll } from '@/api/torrents'

export default defineComponent({
  name: 'TorrentList',
  setup() {
    const torrents = ref<Torrent[]>([])
    const speedData = ref<Map<string, TorrentDynamicInfo>>(new Map())
    const loading = ref(false)
    
    // 定时器
    let staticRefreshTimer: number | null = null
    let dynamicRefreshTimer: number | null = null
    
    // 静态数据：10分钟刷新
    const STATIC_REFRESH_INTERVAL = 10 * 60 * 1000
    
    // 动态数据：10秒刷新
    const DYNAMIC_REFRESH_INTERVAL = 10 * 1000
    
    // 加载静态数据
    const loadStaticData = async () => {
      loading.value = true
      try {
        const response = await getTorrents({ skip: 0, limit: 50 })
        torrents.value = response.data.list
      } finally {
        loading.value = false
      }
    }
    
    // 加载动态数据（速度）
    const loadDynamicSpeed = async () => {
      try {
        const response = await getTorrentsSpeedAll()
        const speedMap = new Map<string, TorrentDynamicInfo>()
        
        response.data.forEach((downloader: any) => {
          downloader.torrents?.forEach((torrent: any) => {
            speedMap.set(torrent.hash, {
              downloadSpeed: torrent.downloadSpeed,
              uploadSpeed: torrent.uploadSpeed,
              num_seeds: torrent.num_seeds,
              num_leechs: torrent.num_leechs
            })
          })
        })
        
        speedData.value = speedMap
      } catch (error) {
        console.error('加载速度数据失败:', error)
      }
    }
    
    // 合并静态和动态数据
    const torrentsWithSpeed = computed(() => {
      return torrents.value.map(torrent => {
        const dynamic = speedData.value.get(torrent.hash)
        return {
          ...torrent,
          downloadSpeed: dynamic?.downloadSpeed ?? torrent.downloadSpeed ?? 0,
          uploadSpeed: dynamic?.uploadSpeed ?? torrent.uploadSpeed ?? 0,
          num_seeds: dynamic?.num_seeds ?? torrent.num_seeds ?? 0,
          num_leechs: dynamic?.num_leechs ?? torrent.num_leechs ?? 0
        }
      })
    })
    
    // 格式化速度显示
    const formatSpeed = (speed: number) => {
      if (speed === 0) return '0 B/s'
      if (speed < 1024) return `${speed} B/s`
      if (speed < 1024 * 1024) return `${(speed / 1024).toFixed(1)} KB/s`
      return `${(speed / 1024 / 1024).toFixed(2)} MB/s`
    }
    
    // 速度颜色
    const getSpeedClass = (speed: number) => {
      if (speed === 0) return 'text-gray'
      if (speed < 1024) return 'text-success'
      if (speed < 1024 * 1024) return 'text-warning'
      return 'text-danger'
    }
    
    onMounted(() => {
      loadStaticData()
      loadDynamicSpeed()
      
      staticRefreshTimer = window.setInterval(loadStaticData, STATIC_REFRESH_INTERVAL)
      dynamicRefreshTimer = window.setInterval(loadDynamicSpeed, DYNAMIC_REFRESH_INTERVAL)
    })
    
    onUnmounted(() => {
      if (staticRefreshTimer) clearInterval(staticRefreshTimer)
      if (dynamicRefreshTimer) clearInterval(dynamicRefreshTimer)
    })
    
    return {
      torrentsWithSpeed,
      loading,
      formatSpeed,
      getSpeedClass
    }
  }
})
</script>

.text-danger { color: #f56c6c; }
.text-gray { color: #909399; }
</style>
```
