#### ✅ 任务1.1: 创建扫描器核心
**文件**: `app/services/orphan_scanner.py`

```python
class OrphanScanner:
    """孤儿文件扫描器"""
    
    def __init__(self, path_mapping_service: PathMappingService = None):
        self.torrent_file_map = {}      # {absolute_path: torrent_info}
        self.seen_inodes = set()        # inode去重（跨平台）
        self.path_mapping_service = path_mapping_service or PathMappingService()
        
    async def auto_discover_scan_paths(
        self, 
        torrents: List[TorrentInfo]
    ) -> List[str]:
        """自动从种子发现扫描路径"""
        scan_roots = set()
        
        for torrent in torrents:
            # 下载器内部路径
            scan_roots.add(torrent.save_path)
            
            # 内容路径（处理 Auto TMM）
            if hasattr(torrent, 'content_path') and torrent.content_path:
                scan_roots.add(torrent.content_path)
        
        # 路径映射转换（内部 → 外部）
        external_roots = []
        for root in scan_roots:
            external_root = self.path_mapping_service.internal_to_external(root)
            if os.path.isabs(external_root):
                external_roots.append(external_root)
        
        return external_roots
    
    def build_torrent_file_map(self, torrents: List[TorrentInfo]) -> None:
        """构建种子文件映射表"""
        for torrent in torrents:
            for file_path in torrent.files:
                abs_path = os.path.abspath(file_path)
                self.torrent_file_map[abs_path] = torrent
    
    async def walk_scan_root(
        self, 
        root: str
    ) -> AsyncIterator[OrphanFile]:
        """遍历扫描根目录"""
        
        for file_path in Path(root).rglob("*"):
            if not file_path.is_file():
                continue
            
            # inode 去重
            try:
                inode_key = InodeHelper.get_file_identifier(str(file_path))
                
                if inode_key in self.seen_inodes:
                    continue
                
                self.seen_inodes.add(inode_key)
            except Exception as e:
                logger.warning(f"获取文件 inode 失败: {file_path}, {e}")
                continue
            
            # 检查是否在种子文件映射中
            abs_path = str(file_path.absolute())
            if abs_path not in self.torrent_file_map:
                yield OrphanFile(
                    path=abs_path,
                    size=file_path.stat().st_size,
                    mtime=file_path.stat().st_mtime
                )
```

#### ✅ 任务1.2: 跨平台 inode 实现
**文件**: `app/services/orphan_scanner.py`

```python
class InodeHelper:
    """跨平台 inode 辅助类"""
    
    @staticmethod
    def get_file_identifier(file_path: str) -> Optional[Tuple]:
        """
        获取文件唯一标识符（跨平台）
        
        Windows: 使用 (device, file_index)
        Linux/Unix: 使用 (st_dev, st_ino)
        """
        try:
            stat_info = os.stat(file_path)
            
            if platform.system() == "Windows":
                return (stat_info.st_dev, stat_info.st_ino)
            else:
                return (stat_info.st_dev, stat_info.st_ino)
                
        except Exception as e:
            logger.warning(f"获取文件标识符失败: {file_path}, {e}")
            return None
```

from fastapi import APIRouter, Depends

router = APIRouter(prefix="/orphans", tags=["orphans"])

@router.get("/latest")
async def get_latest_scan_results():
    """获取最新扫描结果"""
    from app.database import AsyncSessionLocal
    from app.models.orphan_scan import OrphanScanResult
    
    async with AsyncSessionLocal() as db:
        result = await db.execute(
            select(OrphanScanResult)
            .order_by(OrphanScanResult.scan_time.desc())
            .limit(1)
        )
        latest = result.scalar_one_or_none()
        
        if not latest:
            return {"code": "success", "data": None}
        
        return {
            "code": "success",
            "data": {
                "scan_id": latest.scan_id,
                "total_orphans": latest.total_orphans,
                "orphan_list": json.loads(latest.orphan_list)
            }
        }

@router.post("/scan/manual")
async def manual_scan():
    """手动触发扫描"""
    scanner = OrphanScanner()
    result = await scanner.weekly_scan()
    return {"code": "success", "data": result}
