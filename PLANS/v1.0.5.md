```python
from sqlalchemy import Column, Integer, String, Text, Boolean, DateTime, ForeignKey
from sqlalchemy.orm import relationship
from app.database import Base
import json

class QueryTemplate(Base):
    """查询模板"""
    __tablename__ = "query_templates"
    
    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(100), nullable=False, comment="模板名称")
    description = Column(Text, comment="模板描述")
    
    # 查询条件（JSON格式）
    query_conditions = Column(Text, nullable=False, comment="查询条件JSON")
    
    # 分类
    category = Column(String(50), comment="模板分类")
    
    # 权限
    is_public = Column(Boolean, default=False, comment="是否公开")
    is_system = Column(Boolean, default=False, comment="是否系统模板")
    
    # 创建者
    user_id = Column(Integer, ForeignKey("users.id"), comment="创建者ID")
    user = relationship("User", back_populates="query_templates")
    
    # 使用统计
    use_count = Column(Integer, default=0, comment="使用次数")
    
    # 时间戳
    created_at = Column(DateTime, comment="创建时间")
    updated_at = Column(DateTime, comment="更新时间")
    
    def get_conditions(self) -> dict:
        """获取查询条件"""
        return json.loads(self.query_conditions)
    
    def set_conditions(self, conditions: dict):
        """设置查询条件"""
        self.query_conditions = json.dumps(conditions, ensure_ascii=False)
```

#### ✅ 任务1.3: 创建核心服务
**文件**: `app/services/query_template_service.py`

```python
from typing import List, Optional
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models.query_template import QueryTemplate
from app.schemas.query_template import QueryTemplateCreate, QueryTemplateUpdate

class QueryTemplateService:
    """查询模板服务"""
    
    async def create_template(
        self,
        db: AsyncSession,
        user_id: int,
        template_data: QueryTemplateCreate
    ) -> QueryTemplate:
        """创建查询模板"""
        
        # 检查同名模板
        existing = await db.execute(
            select(QueryTemplate).where(
                QueryTemplate.name == template_data.name,
                QueryTemplate.user_id == user_id
            )
        )
        if existing.scalar_one_or_none():
            raise ValueError(f"模板名称已存在: {template_data.name}")
        
        # 创建模板
        db_template = QueryTemplate(
            name=template_data.name,
            description=template_data.description,
            user_id=user_id,
            category=template_data.category,
            is_public=template_data.is_public,
            is_system=False,
            use_count=0
        )
        db_template.set_conditions(template_data.query_conditions)
        
        db.add(db_template)
        await db.commit()
        await db.refresh(db_template)
        
        return db_template
    
    async def get_user_templates(
        self,
        db: AsyncSession,
        user_id: int,
        category: Optional[str] = None
    ) -> List[QueryTemplate]:
        """获取用户的模板列表"""
        
        query = select(QueryTemplate).where(
            (QueryTemplate.user_id == user_id) | 
            (QueryTemplate.is_public == True) |
            (QueryTemplate.is_system == True)
        )
        
        if category:
            query = query.where(QueryTemplate.category == category)
        
        query = query.order_by(QueryTemplate.use_count.desc())
        
        result = await db.execute(query)
        return result.scalars().all()
    
    async def apply_template(
        self,
        db: AsyncSession,
        template_id: int,
        user_id: int
    ) -> Optional[Dict[str, Any]]:
        """应用模板"""
        
        template = await self.get_template(db, template_id, user_id)
        if not template:
            return None
        
        # 获取查询条件
        conditions = template.get_conditions()
        
        # 更新使用次数
        template.use_count += 1
        await db.commit()
        
        return conditions
```

### Phase 2: API 接口 (1天)

#### ✅ 任务2.1: 创建 API 端点
**文件**: `app/api/endpoints/query_templates.py`

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from typing import List
from app.database import get_db
from app.services.query_template_service import QueryTemplateService
from app.schemas.query_template import (
    QueryTemplateCreate,
    QueryTemplateUpdate,
    QueryTemplateResponse
)
from app.auth.dependencies import get_current_user

router = APIRouter(prefix="/query-templates", tags=["query-templates"])
service = QueryTemplateService()

@router.post("/", response_model=QueryTemplateResponse)
async def create_template(
    template_data: QueryTemplateCreate,
    db: AsyncSession = Depends(get_db),
    current_user = Depends(get_current_user)
):
    """创建查询模板"""
    template = await service.create_template(db, current_user.id, template_data)
    return template

@router.get("/", response_model=List[QueryTemplateResponse])
async def list_templates(
    category: str = None,
    db: AsyncSession = Depends(get_db),
    current_user = Depends(get_current_user)
):
    """获取用户的模板列表"""
    templates = await service.get_user_templates(db, current_user.id, category)
    return templates

@router.post("/{template_id}/apply")
async def apply_template(
    template_id: int,
    db: AsyncSession = Depends(get_db),
    current_user = Depends(get_current_user)
):
    """应用模板"""
    conditions = await service.apply_template(db, template_id, current_user.id)
    if not conditions:
        raise HTTPException(status_code=404, detail="模板不存在")
    
    return {
        "code": "success",
        "data": conditions
    }
```
