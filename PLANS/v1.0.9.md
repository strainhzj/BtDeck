#### ✅ 任务1.2: 打包脚本
**文件**: `scripts/build.sh`

```bash
#!/bin/bash

echo "🚀 Building BtDeck v1.0.9..."

# 检查依赖
echo "📦 Checking dependencies..."
python -m pip list | grep -E "(PyInstaller|pyinstaller)"
if [ $? -ne 0 ]; then
    echo "❌ PyInstaller not found. Installing..."
    python -m pip install pyinstaller
fi

# Windows 打包
if [[ "$OSTYPE" == "msys" ]]; then
    echo "📦 Building Windows executable..."
    pyinstaller pyinstaller.spec --onefile --windowed
    echo "✅ Windows build complete: dist/btdeck.exe"
    
# Linux 打包
elif [[ "$OSTYPE" == "linux-gnu"* ]]; then
    echo "📦 Building Linux executable..."
    pyinstaller pyinstaller.spec --onefile
    echo "✅ Linux build complete: dist/btdeck"
    
else
    echo "⚠️  Unsupported platform: $OSTYPE"
fi

# Docker 镜像
echo "🐳 Building Docker image..."
docker build -t btdeck:latest .
echo "✅ Docker build complete: btdeck:latest"

echo "🎉 All builds complete!"
```

### Phase 2: 部署脚本 (2天)

#### ✅ 任务2.1: Linux 安装脚本
**文件**: `scripts/install.sh`

```bash
#!/bin/bash

echo "🔧 Installing BtDeck v1.0.9..."

# 检查 root 权限
if [ "$EUID" -ne 0 ]; then
    echo "❌ Please run as root"
    exit 1
fi

# 创建用户
echo "👤 Creating user..."
useradd -r -s /bin/bash btdeck || true

# 创建目录
echo "📁 Creating directories..."
mkdir -p /opt/btdeck
mkdir -p /var/log/btdeck
mkdir -p /etc/btdeck

# 复制文件
echo "📦 Copying files..."
cp -r . /opt/btdeck/
chown -R btdeck:btdeck /opt/btdeck

# 安装 systemd service
echo "⚙️  Installing systemd service..."
cp scripts/btdeck.service /etc/systemd/system/
systemctl daemon-reload
systemctl enable btdeck

# 安装完成
echo "✅ BtDeck installed successfully!"
echo ""
echo "📝 Quick start:"
echo "  Start:   sudo systemctl start btdeck"
echo "  Stop:    sudo systemctl stop btdeck"
echo "  Status:  sudo systemctl status btdeck"
echo "  Logs:    sudo journalctl -u btdeck -f"
echo ""
echo "🌐 Web interface: http://localhost:5001"
```

**验收标准**:
- [ ] 用户创建正确

#### ✅ 任务2.2: Windows 安装脚本
**文件**: `scripts/install.bat`

```batch
@echo off
echo 🔧 Installing BtDeck v1.0.9...

REM 创建目录
echo 📁 Creating directories...
if not exist "C:\BtDeck" mkdir "C:\BtDeck"
if not exist "C:\BtDeck\logs" mkdir "C:\BtDeck\logs"

REM 复制文件
echo 📦 Copying files...
xcopy /E /I dist\* "C:\BtDeck\"

REM 安装服务（需要 NSSM）
echo ⚙️  Installing Windows service...
nssm install BtDeck "C:\BtDeck\btdeck.exe"
nssm set BtDeck AppDirectory "C:\BtDeck"
nssm set BtDeck AppParameters -- "-m uvicorn app.main:app --host 0.0.0.0 --port 5001"
nssm set BtDeck AppStdout "C:\BtDeck\logs\stdout.log"
nssm set BtDeck AppStderr "C:\BtDeck\logs\stderr.log"

REM 启动服务
nssm start BtDeck

echo ✅ BtDeck installed successfully!
echo.
echo 📝 Quick start:
echo   Start:   nssm start BtDeck
echo   Stop:    nssm stop BtDeck
echo   Status:  nssm status BtDeck
echo.
echo 🌐 Web interface: http://localhost:5001
```

### Phase 3: 容器化部署 (1-2天)

#### ✅ 任务3.1: Dockerfile 优化
**文件**: `Dockerfile`

```dockerfile
# Multi-stage build for smaller image
FROM python:3.11-slim as builder

WORKDIR /app

# 安装系统依赖
RUN apt-get update && apt-get install -y \
    gcc \
    g++ \
    && rm -rf /var/lib/apt/lists/*

# 安装 Python 依赖
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 复制应用代码
COPY . .

# Production stage
FROM python:3.11-slim

WORKDIR /app

# 创建非 root 用户
RUN useradd -m -u 1000 btdeck && \
    chown -R btdeck:btdeck /app

# 复制构建产物
COPY --from=builder /usr/local/lib/python3.11/site-packages /usr/local/lib/python3.11/site-packages
COPY --from=builder /app /app

# 创建日志目录
RUN mkdir -p /var/log/btdeck && \
    chown -R btdeck:btdeck /var/log/btdeck

# 切换到非 root 用户
USER btdeck

# 暴露端口
EXPOSE 5001

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python -c "import requests; requests.get('http://localhost:5001/docs').status_code == 200"

# 启动命令
CMD ["python", "-m", "uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "5001"]
```

#### ✅ 任务3.2: Docker Compose 配置
**文件**: `docker-compose.yml`

```yaml
version: '3.8'

services:
  btdeck:
    build: .
    container_name: btdeck
    ports:
      - "5001:5001"
    environment:
      - DB_TYPE=postgresql
      - DB_HOST=postgres
      - DB_PORT=5432
      - DB_USER=btdeck
      - DB_PASSWORD=btdeck_password
      - DB_NAME=btdeck
    volumes:
      - btdeck_data:/app/data
      - btdeck_logs:/var/log/btdeck
    depends_on:
      - postgres
    restart: unless-stopped
    networks:
      - btdeck_network
  
  postgres:
    image: postgres:15-alpine
    container_name: btdeck_postgres
    environment:
      - POSTGRES_USER=btdeck
      - POSTGRES_PASSWORD=btdeck_password
      - POSTGRES_DB=btdeck
    volumes:
      - postgres_data:/var/lib/postgresql/data
    restart: unless-stopped
    networks:
      - btdeck_network

volumes:
  btdeck_data:
  btdeck_logs:
  postgres_data:

networks:
  btdeck_network:
    driver: bridge
```
