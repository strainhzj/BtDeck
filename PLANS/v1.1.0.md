**文件**: `app/services/automation_service.py`

```python
from typing import Dict, Any, Callable, Optional
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class AutomationService:
    """自动化服务 - 管理所有自动化任务"""
    
    def __init__(self):
        self._tasks = {}  # {task_name: task_config}
        self._schedules = {}  # {task_name: schedule_config}
        
    def register_task(
        self,
        name: str,
        func: Callable,
        description: str,
        default_interval: int,
        enabled: bool = True
    ):
        """注册自动化任务"""
        self._tasks[name] = {
            "func": func,
            "description": description,
            "enabled": enabled,
            "default_interval": default_interval,
            "last_run": None,
            "next_run": None,
            "run_count": 0,
            "success_count": 0,
            "failure_count": 0
        }
        
        logger.info(f"注册自动化任务: {name}")
    
    def schedule_task(
        self,
        name: str,
        interval: int,
        enabled: bool = True
    ):
        """调度任务"""
        if name not in self._tasks:
            raise ValueError(f"任务不存在: {name}")
        
        self._schedules[name] = {
            "interval": interval,
            "enabled": enabled
        }
        
        logger.info(f"调度任务: {name}, 间隔: {interval}秒")
    
    async def run_task(self, name: str) -> Dict[str, Any]:
        """执行任务"""
        if name not in self._tasks:
            return {"status": "error", "message": f"任务不存在: {name}"}
        
        task = self._tasks[name]
        
        if not task["enabled"]:
            return {"status": "skipped", "message": "任务未启用"}
        
        try:
            logger.info(f"执行自动化任务: {name}")
            
            # 执行任务
            result = await task["func"]()
            
            # 更新统计
            task["last_run"] = datetime.now()
            task["run_count"] += 1
            task["success_count"] += 1
            
            logger.info(f"任务执行成功: {name}")
            
            return {
                "status": "success",
                "task_name": name,
                "result": result,
                "run_count": task["run_count"]
            }
            
        except Exception as e:
            task["failure_count"] += 1
            logger.error(f"任务执行失败: {name}, 错误: {e}")
            
            return {
                "status": "error",
                "task_name": name,
                "error": str(e)
            }
    
    def get_task_status(self, name: str) -> Dict[str, Any]:
        """获取任务状态"""
        if name not in self._tasks:
            return {"error": "任务不存在"}
        
        task = self._tasks[name]
        schedule = self._schedules.get(name, {})
        
        return {
            "name": name,
            "description": task["description"],
            "enabled": task["enabled"],
            "last_run": task["last_run"].isoformat() if task["last_run"] else None,
            "run_count": task["run_count"],
            "success_count": task["success_count"],
            "failure_count": task["failure_count"],
            "interval": schedule.get("interval"),
            "schedule_enabled": schedule.get("enabled", False)
        }
```

#### ✅ 任务1.2: 实现自动化任务
**文件**: `app/tasks/automation/auto_cleanup_orphans.py`

```python
"""
自动清理孤儿文件任务（每周执行）
"""
from app.services.automation_service import AutomationService
from app.services.orphan_scanner import OrphanScanner

async def auto_cleanup_orphans():
    """自动清理孤儿文件
    
    策略:
    - 删除超过30天的孤儿文件
    - 删除大小为0的文件
    - 保留正在下载的文件
    """
    scanner = OrphanScanner()
    
    # 1. 扫描孤儿文件
    scan_result = await scanner.weekly_scan()
    
    if not scan_result or scan_result.get("total_orphans", 0) == 0:
        return {"message": "无孤儿文件需要清理"}
    
    # 2. 筛选可删除的文件
    import time
    current_time = time.time()
    thirty_days_ago = current_time - (30 * 24 * 3600)
    
    orphan_list = scan_result.get("orphan_list", [])
    to_delete = []
    
    for orphan in orphan_list:
        # 删除条件: 超过30天 或 大小为0
        if orphan["mtime"] < thirty_days_ago or orphan["size"] == 0:
            to_delete.append(orphan["path"])
    
    # 3. 执行删除
    deleted_count = 0
    for path in to_delete:
        try:
            import os
            os.remove(path)
            deleted_count += 1
        except Exception as e:
            logger.warning(f"删除文件失败: {path}, {e}")
    
    return {
        "total_orphans": len(orphan_list),
        "deleted_count": deleted_count,
        "skipped_count": len(orphan_list) - deleted_count
    }

# 注册任务
automation = AutomationService()
automation.register_task(
    name="auto_cleanup_orphans",
    func=auto_cleanup_orphans,
    description="自动清理孤儿文件（每周）",
    default_interval=7 * 24 * 3600  # 7天
)
```

#### ✅ 任务2.1: API 接口
**文件**: `app/api/endpoints/automation.py`

```python
from fastapi import APIRouter, Depends
from app.services.automation_service import AutomationService

router = APIRouter(prefix="/automation", tags=["automation"])

# 全局自动化服务实例
automation_service = AutomationService()

@router.get("/tasks")
async def list_automation_tasks():
    """列出所有自动化任务"""
    return {
        "code": "success",
        "data": automation_service.list_tasks()
    }

@router.get("/tasks/{task_name}")
async def get_task_status(task_name: str):
    """获取任务状态"""
    return {
        "code": "success",
        "data": automation_service.get_task_status(task_name)
    }

@router.post("/tasks/{task_name}/run")
async def run_task_manually(task_name: str):
    """手动执行任务"""
    result = await automation_service.run_task(task_name)
    return {
        "code": "success" if result["status"] == "success" else "error",
        "data": result
    }

@router.put("/tasks/{task_name}/schedule")
async def update_task_schedule(
    task_name: str,
    interval: int,
    enabled: bool = True
):
    """更新任务调度"""
    try:
        automation_service.schedule_task(task_name, interval, enabled)
        return {
            "code": "success",
            "message": f"任务 {task_name} 调度已更新"
        }
    except Exception as e:
        return {
            "code": "error",
            "message": str(e)
        }
```

```vue
<template>
  <div class="automation-container">
    <h2>自动化任务</h2>
    
    <!-- 任务列表 -->
    <el-table :data="tasks" v-loading="loading">
      <el-table-column prop="name" label="任务名称" />
      <el-table-column prop="description" label="描述" />
      <el-table-column prop="enabled" label="状态">
        <template #default="{ row }">
          <el-tag :type="row.enabled ? 'success' : 'info'">
            {{ row.enabled ? '启用' : '禁用' }}
          </el-tag>
        </template>
      </el-table-column>
      <el-table-column prop="last_run" label="上次运行">
        <template #default="{ row }">
          {{ row.last_run || '从未运行' }}
        </template>
      </el-table-column>
      <el-table-column prop="run_count" label="运行次数" />
      <el-table-column label="操作">
        <template #default="{ row }">
          <el-button size="small" @click="handleRun(row.name)">
            立即运行
          </el-button>
        </template>
      </el-table-column>
    </el-table>
  </div>
</template>

<script lang="ts">
import { defineComponent, ref, onMounted } from 'vue'
import { getAutomationTasks, runAutomationTask } from '@/api/automation'

export default defineComponent({
  name: 'AutomationIndex',
  setup() {
    const tasks = ref([])
    const loading = ref(false)
    
    const loadTasks = async () => {
      loading.value = true
      try {
        const response = await getAutomationTasks()
        tasks.value = response.data
      } finally {
        loading.value = false
      }
    }
    
    const handleRun = async (taskName: string) => {
      try {
        await runAutomationTask(taskName)
        ElMessage.success('任务已启动')
        await loadTasks()
      } catch (error) {
        ElMessage.error('任务启动失败')
      }
    }
    
    onMounted(() => {
      loadTasks()
    })
    
    return {
      tasks,
      loading,
      handleRun
    }
  }
})
</script>
```

**验收标准**:
